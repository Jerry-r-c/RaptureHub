local function setVelocity(hrp, vec)
    if hrp.AssemblyLinearVelocity ~= nil then
        hrp.AssemblyLinearVelocity = vec
    else
        hrp.Velocity = vec
    end
end

local function getVelocity(hrp)
    if hrp.AssemblyLinearVelocity ~= nil then
        return hrp.AssemblyLinearVelocity
    else
        return hrp.Velocity
    end
end

-- // Enhanced Draggable Logic (PC & Mobile)
local dragToggle, dragStart, startPos

local function updateInput(input)
    local delta = input.Position - dragStart
    local position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    TweenService:Create(Frame, TweenInfo.new(0.10), {Position = position}):Play()
end

Frame.InputBegan:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
        dragToggle = true
        dragStart = input.Position
        startPos = Frame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragToggle = false
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragToggle and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateInput(input)
    end
end)

-- // Logic Variables
local isEnabled = false
local isFlicking = false
local lastFlickTime = 0
local lastHitInstance = nil
local Camera = workspace.CurrentCamera

local function setEnabled(state)
    isEnabled = state

    local targetColor = state and Color3.fromRGB(80, 255, 80) or Color3.fromRGB(255, 60, 60)
    TweenService:Create(Dot, TweenInfo.new(0.2), {BackgroundColor3 = targetColor}):Play()

    ToggleBtn.TextColor3 = state and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(200, 200, 200)
    SubLabel.Text = state and "ON - detecting walls" or "OFF - tap to enable"
    SubLabel.TextColor3 = state and Color3.fromRGB(80, 200, 80) or Color3.fromRGB(100, 100, 100)

    local strokeTarget = state and Color3.fromRGB(80, 200, 80) or Color3.fromRGB(60, 60, 60)
    TweenService:Create(Stroke, TweenInfo.new(0.25), {Color = strokeTarget}):Play()
end

local function performFlick()
    if isFlicking then return end
    isFlicking = true

    local char = LocalPlayer.Character
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then
        isFlicking = false
        return
    end

    hum:ChangeState(Enum.HumanoidStateType.Jumping)

    local vel = getVelocity(hrp)
    setVelocity(hrp, Vector3.new(vel.X, 50, vel.Z))

    local originalCFrame = Camera.CFrame
    Camera.CFrame = originalCFrame * CFrame.Angles(0, math.rad(180), 0)
    task.wait(0.01)
    Camera.CFrame = originalCFrame

    isFlicking = false
end

RunService.Heartbeat:Connect(function()
    if not isEnabled then return end

    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local params = RaycastParams.new()
    params.FilterDescendantsInstances = {char}
    params.FilterType = Enum.RaycastFilterType.Exclude

    local result = workspace:Raycast(hrp.Position, Camera.CFrame.LookVector * 3, params)

    if result and result.Instance and result.Instance.CanCollide then
        if lastHitInstance and lastHitInstance ~= result.Instance then
            local now = tick()
            if now - lastFlickTime > 0.05 then
                lastFlickTime = now
                performFlick()
            end
        end
        lastHitInstance = result.Instance
    else
        lastHitInstance = nil
    end
end)

ToggleBtn.Activated:Connect(function()
    setEnabled(not isEnabled)
end)
